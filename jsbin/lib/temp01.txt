function solution(num)  {
  let answer = "";
  const dfs = (level) => {
    if (level === 0) return;
    else {
      dfs(Math.floor(level / 2));
      answer += level % 2;
    }
  };
  dfs(num);
  
  return ('0'.repeat(32) + answer).substr(-32);
}
